const { REST, Routes, SlashCommandBuilder } = require('discord.js');
const commands = [
  new SlashCommandBuilder()
    .setName('addpoints')
    .setDescription('إضافة نقاط لعضو')
    .addUserOption(opt => opt.setName('user').setDescription('العضو').setRequired(true))
    .addIntegerOption(opt => opt.setName('amount').setDescription('عدد النقاط').setRequired(true))
    .toJSON(),

  new SlashCommandBuilder()
    .setName('removepoints')
    .setDescription('سحب نقاط من عضو')
    .addUserOption(opt => opt.setName('user').setDescription('العضو').setRequired(true))
    .addIntegerOption(opt => opt.setName('amount').setDescription('عدد النقاط').setRequired(true))
    .toJSON(),

  new SlashCommandBuilder()
    .setName('points')
    .setDescription('عرض نقاط عضو')
    .addUserOption(opt => opt.setName('user').setDescription('العضو'))
    .toJSON(),
];

const rest = new REST({ version: '10' }).setToken(process.env.TOKEN);

(async () => {
  try {
    console.log('🔁 Registering commands...');
    // لو تبغى تسجّل في سيرفر محدد أثناء التجربة: استخدم GUILD_ID
    if (process.env.GUILD_ID) {
      await rest.put(
        Routes.applicationGuildCommands(process.env.CLIENT_ID, process.env.GUILD_ID),
        { body: commands },
      );
      console.log('✅ Guild commands registered.');
    } else {
      await rest.put(
        Routes.applicationCommands(process.env.CLIENT_ID),
        { body: commands },
      );
      console.log('✅ Global commands registered (may take up to 1 hour to appear).');
    }
  } catch (err) {
    console.error(err);
  }
})();
